// 现在（从小 → 大）
const t = i / this.points.length;
const width = 8 * (1 - t);

// 改成（从大 → 小）
const t = i / this.points.length;
const width = 8 * t;

// Smoothstep（非常自然）
const t = i / (this.points.length - 1);
const k = 1 - t;
const width = 10 * (k * k * (3 - 2 * k)); // smoothstep

// 指数型（速度感更强）
const t = i / (this.points.length - 1);
const width = 10 * Math.pow(1 - t, 1.8);
